
local CHANNEL_NOTES = 2

local count = ll.LinksetDataCountKeys()

local function addNote(note)
    count += 1
    ll.LinksetDataWrite(tostring(count), note)
    ll.OwnerSay("Note added: \n" .. note)
end

local function sayNotes()
    ll.OwnerSay("---" .. count .. " quick notes ---")
    for i = 1, count do
        ll.OwnerSay(ll.LinksetDataRead(tostring(i)))
    end
end

function LLEvents.listen(channel, name, id, message)
    if channel == CHANNEL_NOTES then
        if message == "RESET" then
            sayNotes()
            ll.LinksetDataReset()
            count = 0
            ll.OwnerSay("\nNotes deleted.")
        else
            addNote(message)
        end
    end
end

function LLEvents.touch_start(events)
    if events[1]:getKey() == ll.GetOwner() then
        sayNotes()
    end
end

function LLEvents.changed(change)
    if bit32.btest(change, CHANGED_OWNER) then
        ll.LinksetDataReset()
        ll.ResetScript()
    end
end

ll.Listen(CHANNEL_NOTES, "", ll.GetOwner(), "")

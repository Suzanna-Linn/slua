
Animal = {}
Animal.__index = Animal

function Animal:new(name, age)
    return setmetatable({
        name = name or "Unknown",
        age = age or 0,
    }, self)
end

function Animal:speak()
    print(self.name .. " makes a generic animal sound.")
end

function Animal:introduce()
    print("Hi, I am " .. self.name .. " and I am " .. self.age .. " years old.")
end

function Animal:__tostring()
    return "[Animal] Name: " .. self.name .. ", Age: " .. self.age
end

-- ----------

Horse = {}
Horse.__index = Horse
setmetatable(Horse, { __index = Animal }) 

function Horse:new(name, age, color)
    local instance = Animal.new(self, name, age)
    instance.color = color
    return instance
end

function Horse:speak()
    print(self.name .. " says: Neigh!")
end

function Horse:gallop()
    print(self.name .. " is galloping majestically in the wind.")
end

function Horse:__tostring()
    return "[Horse] " .. self.name .. " is a " .. self.color .. " horse."
end

-- ----------

Eagle = {}
Eagle.__index = Eagle
setmetatable(Eagle, {__index = Animal})

function Eagle:new(name, age, wingspan)
    local instance = Animal.new(self, name, age)
    instance.wingspan = wingspan
    return instance
end

function Eagle:speak()
    print(self.name .. " says: Screeeech!")
end

function Eagle:fly()
    print(self.name .. " is flying high with a wingspan of " .. self.wingspan .. "m.")
end

function Eagle:__tostring()
    return "[Eagle] " .. self.name .. " (Wingspan: " .. self.wingspan .. "m)"
end

-- ----------

Pegasus = {}
Pegasus.__index = Pegasus
setmetatable(Pegasus, { __index = function(t, key)
    return Horse[key] or Eagle[key]
end } )

function Pegasus:new(name, age, color, wingspan, aura)
    local instance = Horse.new(self, name, age, color)
    for k, v in Eagle.new(self, name, age, wingspan) do
        if not instance[k] then
            instance[k] = v
        end
    end
    instance.aura = aura or "Golden" 
    return instance
end

function Pegasus:ascendToOlympus()
    print(self.name .. " glows with a " .. self.aura .. " aura and ascends to Mount Olympus!")
end

function Pegasus:speak()
    print(self.name .. " neighs and screeches simultaneously!")
end

function Pegasus:__tostring()
    return "[Pegasus] " .. self.name .. " (Aura: " .. self.aura .. ")"
end

-- ----------



print("--- 1. INSTANTIATION ---")
local a = Animal:new("Generic Beast", 10)
local h = Horse:new("Spirit", 5, "Mustang color")
local e = Eagle:new("Freedom", 3, 2.5)
local p = Pegasus:new("Hercules", 1000, "White", 10.0, "Celestial")

print(a) -- > [Animal] Name: Generic Beast, Age: 10
print(h) -- > [Horse] Spirit is a Mustang color horse.
print(e) -- > [Eagle] Freedom (Wingspan: 2.5m)
print(p) -- > [Pegasus] Hercules (Aura: Celestial)


print("\n--- 2. INHERITANCE (introduce) ---")
-- All classes inherit 'introduce' from Animal (Pegasus finds it via Horse)
a:introduce() -- > Hi, I am Generic Beast and I am 10 years old.
h:introduce() -- > Hi, I am Spirit and I am 5 years old.
e:introduce() -- > Hi, I am Freedom and I am 3 years old.
p:introduce() -- > Hi, I am Hercules and I am 1000 years old.


print("\n--- 3. POLYMORPHISM (speak) ---")
-- Each class has its own version of 'speak'
a:speak() -- > Generic Beast makes a generic animal sound.
h:speak() -- > Spirit says: Neigh!
e:speak() -- > Freedom says: Screeeech!
p:speak() -- > Hercules neighs and screeches simultaneously!


print("\n--- 4. SPECIALIZED METHODS ---")
-- Testing methods unique to specific classes
h:gallop() -- > Spirit is galloping majestically in the wind.
e:fly()    -- > Freedom is flying high with a wingspan of 2.5m.

-- Pegasus can do both because of Multiple Inheritance
p:gallop() -- > Hercules is galloping majestically in the wind. (Inherited from Horse)
p:fly()    -- > Hercules is flying high with a wingspan of 10m. (Inherited from Eagle)


print("\n--- 5. MYTHOLOGIC METHODS ---")
-- Testing the unique method added only to Pegasus
p:ascendToOlympus()  -- > Hercules glows with a Celestial aura and ascends to Mount Olympus!


print("\n--- 6. ERROR CHECKING ---")
print(select(2, pcall(h.fly, h)))             -- attempt to call method 'fly' (a nil value)
print(select(2, pcall(e.gallop, e)))          -- attempt to call method 'gallop' (a nil value)
print(select(2, pcall(h.ascendToOlympus, h))) -- Regular horses cannot ascend to Olympus (attempt to call a nil value)

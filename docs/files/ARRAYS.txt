local function arrayLen(tab)  -- length of the correlative part of the array
    local len = #tab
    for i = 1, len do
        if tab[i] == nil then
            len = i - 1
            break
        end
    end
    return len
end

local function arrayHasNils(tab)  -- are there nils in the array?
    local hasNils = false
    for i = 1, #tab do
        if tab[i] == nil then
            hasNils = true
            break
        end
    end
    return hasNils
end

local function arrayOnly(tab)  -- is an array with only correlative elements and nothing else?
    local len = #tab
    if len > 0 and next(tab, len) == nil then
        for i = 1, #tab do
            if tab[i] == nil then
                return false
            end
        end
        return true
    else
        return false
    end
end

print(arrayLen(x))  -- 2
print(arrayHasNils(x))  -- true
print(arrayOnly(x))  -- false

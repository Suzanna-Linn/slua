-- with a LL function

-- with a function
local function greet(name)
    ll.Say(0, "hello " .. name)
end

greet("Suzanna")  -- > hello Suzanna



-- with an event
local function greet(id)
    ll.Say(0, "hello " .. ll.GetDisplayName(id))
end

function touch_start(num_detected)
    greet(ll.DetectedKey(0))
end
-- > hello Suzanna



-- with a function returning a value
local function getFullName(id)
    local username = ll.GetUsername(id)
    local displayName = ll.GetDisplayName(id)
    return displayName .. " (" .. username .. ")"
end

local function greet(id)
    ll.Say(0, "hello " .. getFullName(id))
end

function touch_start(num_detected)
    greet(ll.DetectedKey(0))
end
-- > hello Suzanna (suzannalinn)



-- with if-else
local function getFullName(id)
    local username = ll.GetUsername(id)
    local displayName = ll.GetDisplayName(id)
    if ll.ToLower(displayName) == username then
        return displayName
    else
        return displayName .. " (" .. username .. ")"
    end
end

local function greet(id)
    ll.Say(0, "hello " .. getFullName(id))
end

function touch_start(num_detected)
    greet(ll.DetectedKey(0))
end
-- > hello Suzanna (suzannalinn)



-- with inner function
local function greet(id)

    local function getFullName(id)
        local username = ll.GetUsername(id)
        local displayName = ll.GetDisplayName(id)
        if ll.ToLower(displayName) == username then
            return displayName
        else
            return displayName .. " (" .. username .. ")"
        end
    end

    ll.Say(0, "hello " .. getFullName(id))
end

function touch_start(num_detected)
    greet(ll.DetectedKey(0))
end
-- > hello Suzanna (suzannalinn)

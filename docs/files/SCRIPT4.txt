local colors = { COLOR.WHITE, COLOR.GREEN, COLOR.RED }
local texts = { "Still updating inventoryâ€¦", "Lag is my natural habitat.", "Contains traces of prims.", "Mesh? More like mess." }
local alpha = 1.0

local currentColor = 1
local currentText = 1

local function showText()
    local text = texts[currentText]
    local color = colors[currentColor]
    ll.SetText(text, color, alpha)
end

local function color()
    showText()
    currentColor += 1
    if currentColor > #colors then
        currentColor = 1
    end
end

local function text()
    currentText += 1
    if currentText > #texts then
        currentText = 1
    end
end

LLTimers:every(0.5, color)
LLTimers:every(15, text)


-- user parameters, you can change them

local DISPLAY_SECONDS = 5  -- seconds with the slide fully displayed

-- script parameters, don't change

local SLIDE_FACE = 4  -- face number
local ROT = 0         -- rotation in radians

local Slides = {}
local TotalSlides = 0
local CurrentSlide = 1

local function GetSlides()  -- get the slides from the contents at the start, they will be displayed in alphabetical order
    for Texture = 1, ll.GetInventoryNumber(INVENTORY_TEXTURE) do
        local TextureName = ll.GetInventoryName(INVENTORY_TEXTURE, Texture)
        Slides = table.insert(Slides, TextureName)
    end
    TotalSlides = #Slides
end

local function ShowSlide()  -- show one slide
    local SlideName = Slides[CurrentSlide]
    ll.SetPrimitiveParams({PRIM_TEXTURE, SLIDE_FACE, SlideName, vector(1, 1, 0), vector(0, 0, 0), ROT})
    CurrentSlide = if CurrentSlide < TotalSlides then currentSlide + 1 else 1
end

function LLEvents.on_rez(start_param)
    ll.ResetScript()
end

function LLEvents.changed(change)
    if bit32.btest(change, bit32.bor(CHANGED_OWNER, CHANGED_INVENTORY)) then
        ll.ResetScript()
    end
end

GetSlides()
if TotalSlides > 0 then
    ShowSlide()
    LLTimers:on(DISPLAY_SECONDS, ShowSlide)
else
    ll.OwnerSay("no textures to view, add some textures to the viewer contents")
end

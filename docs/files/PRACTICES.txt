-- maximum of two numbers
local function maxOfTwo(a, b)
    if a > b then
        return a
    else
        return b
    end
end

print(maxOfTwo(4, 2))  -- > 4
print(maxOfTwo(10, 25))  -- > 25


-- positive, negative or zero
local function numberSign(n)
    if n > 0 then
        return "positive"
    elseif n < 0 then
        return "negative"
    else
        return "zero"
    end
end

print(numberSign(5))  -- > positive
print(numberSign(-3))  -- > negative
print(numberSign(0))  -- > zero

-- guess the number, give hints and say the tries
local guess = ll.GetWallclock()
local tries = 0

local function checkNumber(num)
    tries = tries + 1
    if num == guess then
	print("Guessed! You have got it in " .. tries .. " tries")
    elseif num > guess then
        print("The number to guess is smaller")
    else
        print("The number to guess is bigger")
    end
end

ll.Listen(1, "", ll.GetOwner(), "")

function listen(channel, name, id, message)
    checkNumber(tonumber(message))
end

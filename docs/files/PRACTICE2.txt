
Exercise2 = [[ The Light Switch
Create a LightSwitch class.
The new function should create a switch that is initially off.
Add a toggle() method which flips the state of the switch (if it's on, turn it off; if it's off, turn it on).
Add a getState() method that prints a message to the console, like "The light is currently on.".
Allow two LightSwitch objects to be compared with the == operator. Two switches should be considered equal if and only if their states are the same.
]]

LightSwitch = {}
LightSwitch.__index = LightSwitch

function LightSwitch:new()
  local instance = {
    isOn = false
  }
  setmetatable(instance, self)
  return instance
end

function LightSwitch:toggle()
  self.isOn = not self.isOn
end

function LightSwitch:getState()
  print(`The light is currently {if self.isOn then "on" else "off"}.`)
end

function LightSwitch:__tostring()
  return `LightSwitch[{if self.isOn then "ON" else "OFF"}]`
end

function LightSwitch.__eq(left, right)
  if getmetatable(left) == LightSwitch and getmetatable(right) == LightSwitch then
    return left.isOn == right.isOn
  end
  return false
end

-- --- Example Usage ---

local kitchenLight = LightSwitch:new()

kitchenLight:getState() -- > The light is currently off.
kitchenLight:toggle()
kitchenLight:getState() -- > The light is currently on.
kitchenLight:toggle()
kitchenLight:getState() -- > The light is currently off.

print("Kitchen light status:", kitchenLight) -- > Kitchen light status:	LightSwitch[OFF]
kitchenLight:toggle()
print("Kitchen light status:", kitchenLight) -- > Kitchen light status:	LightSwitch[ON]

local livingRoomLight = LightSwitch:new()
local bedroomLight = LightSwitch:new()

print("livingRoomLight == bedroomLight:", livingRoomLight == bedroomLight) -- > livingRoomLight == bedroomLight: true
print("kitchenLight == livingRoomLight:", kitchenLight == livingRoomLight) -- > kitchenLight == livingRoomLight: false

bedroomLight:toggle()
print("Bedroom light is now:", bedroomLight) -- > Bedroom light is now:	LightSwitch[ON]
print("kitchenLight == bedroomLight:", kitchenLight == bedroomLight) -- > kitchenLight == bedroomLight:	true

print("kitchenLight == 'hello':", kitchenLight == "hello") -- > kitchenLight == 'hello': false

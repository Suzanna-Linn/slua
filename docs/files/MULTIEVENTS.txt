
-- old SLua

function touch_start(num_detected)
    for i = 0, num_detected -1 do
        local toucher = ll.DetectedKey(i)
        print(ll.DetectedName(i))
    end
end


-- SLua Beta

function LLEvents.touch_start(events)
    for _, ev in events do
        local toucher = ev:getKey()
        print(ev:getName())
    end
end
